 88async def listen(connection_cb: ConnectionCb,
 89                 request_cb: RequestCb,
 90                 addr: Address = Address('0.0.0.0', 102)
 91                 ) -> 'Server':
 92    """Create MMS listening server
 93
 94    Args:
 95        connection_cb: new connection callback
 96        request_cb: received request callback
 97        addr: local listening address
 98
 99    """
100
101    async def on_validate(syntax_names, user_data):
102        syntax_name, req_user_data = user_data
103        if syntax_name != _mms_syntax_name:
104            raise Exception('invalid mms syntax name')
105        initiate_req = _decode(req_user_data)
106        if initiate_req[0] != 'initiate-RequestPDU':
107            raise Exception('invalid initiate request')
108        initiate_res = 'initiate-ResponsePDU', {
109            'negotiatedMaxServOutstandingCalling': 5,
110            'negotiatedMaxServOutstandingCalled': 5,
111            'negotiatedDataStructureNestingLevel': 4,  # TODO compatibility
112            'initResponseDetail': {
113                'negotiatedVersionNumber': 1,
114                'negotiatedParameterCBB': _parameter_cbb,
115                'servicesSupportedCalled': _service_support}}
116        if 'localDetailCalling' in initiate_req[1]:
117            initiate_res[1]['localDetailCalled'] = \
118                initiate_req[1]['localDetailCalling']
119        res_user_data = _encode(initiate_res)
120        return _mms_syntax_name, res_user_data
121
122    async def on_connection(acse_conn):
123        try:
124            try:
125                conn = _create_connection(request_cb, acse_conn)
126            except Exception:
127                await aio.uncancellable(acse_conn.async_close())
128                raise
129            try:
130                await aio.call(connection_cb, conn)
131            except BaseException:
132                await aio.uncancellable(conn.async_close())
133                raise
134        except Exception as e:
135            mlog.error("error creating new incomming connection: %s",
136                       e, exc_info=e)
137
138    async def wait_acse_server_closed():
139        try:
140            await acse_server.wait_closed()
141        finally:
142            async_group.close()
143

146    async_group.spawn(aio.call_on_cancel, acse_server.async_close)
147    async_group.spawn(wait_acse_server_closed)

151    srv._acse_server = acse_server

189

204    async def receive_unconfirmed(self) -> common.Unconfirmed:
205        """Receive unconfirmed message
206
207        Raises:
208            ConnectionError: in case connection is not open
209
210        """
211        try:
212            return await self._unconfirmed_queue.get()
213        except aio.QueueClosedError:
214            raise ConnectionError('connection is not open')
215
216    def send_unconfirmed(self, unconfirmed: common.Unconfirmed):
217        """Send unconfirmed message"""
218        pdu = 'unconfirmed-PDU', {
219            'service': encoder.encode_unconfirmed(unconfirmed)}
220        data = _mms_syntax_name, _encode(pdu)
221        self._acse_conn.write(data)
222
223    async def send_confirmed(self,
224                             req: common.Request
225                             ) -> common.Response:
226        """Send confirmed request and wait for response
227
228        Raises:
229            ConnectionError: in case connection is not open
230
231        """
232        if self._async_group.is_closing:
233            raise ConnectionError('connection is not open')
234        invoke_id = self._last_invoke_id + 1
235        pdu = 'confirmed-RequestPDU', {
236            'invokeID': invoke_id,
237            'service': encoder.encode_request(req)}
238        data = _mms_syntax_name, _encode(pdu)
239        self._acse_conn.write(data)
240        self._last_invoke_id = invoke_id
241        self._response_futures[invoke_id] = asyncio.Future()
242        try:
243            return await self._response_futures[invoke_id]
244        finally:
245            del self._response_futures[invoke_id]
246
247    async def _read_loop(self):
248        running = True
249        try:
250            while running:
251                syntax_name, entity = await self._acse_conn.read()
252                if syntax_name != _mms_syntax_name:
253                    continue
254                pdu = _decode(entity)
255                running = await self._process_pdu(pdu)
256        except asyncio.CancelledError:
257            pdu = 'conclude-RequestPDU', None
258            data = _mms_syntax_name, _encode(pdu)
259            self._acse_conn.write(data)
260            # TODO: wait for response
261            raise
262        finally:
263            self._async_group.close()
264            self._unconfirmed_queue.close()
265            for response_future in self._response_futures.values():
266                if not response_future.done():
267                    response_future.set_exception(
268                        ConnectionError('connection is not open'))
269            await aio.uncancellable(self._acse_conn.async_close())
270
271    async def _process_pdu(self, pdu):
272        name, data = pdu
273
274        if name == 'unconfirmed-PDU':
275            unconfirmed = encoder.decode_unconfirmed(data['service'])
276            await self._unconfirmed_queue.put(unconfirmed)
277            return True
278
279        elif name == 'confirmed-RequestPDU':
280            invoke_id = data['invokeID']
281            req = encoder.decode_request(data['service'])
282            res = await aio.call(self._request_cb, self, req)
283            if isinstance(res, common.ErrorResponse):
284                res_pdu = 'confirmed-ErrorPDU', {
285                    'invokeID': invoke_id,
286                    'serviceError': {
287                        'errorClass': (res.error_class.value, res.value)}}
288            else:
289                res_pdu = 'confirmed-ResponsePDU', {
