243    async def _send(self, device_id, req):
244        req_pdu = messages.Pdu(fc=_get_req_fc(req), data=req)
245        future = asyncio.Future()
246
247        try:
248            self._send_queue.put_nowait((device_id, req_pdu, future))
249
250        except aio.QueueClosedError:
251            raise ConnectionError()
252
253        res_pdu = await future
254        if res_pdu.fc != req_pdu.fc:
255            raise Exception("invalid response function code")
256
257        return res_pdu.data

259    async def _send_loop(self):
260        transaction_id = 0
261        future = None
262
263        try:
264            while True:
265                device_id, req_pdu, future = await self._send_queue.get()
266                if future.done():
267                    continue
268
269                if self._modbus_type == common.ModbusType.TCP:
270                    transaction_id = (transaction_id + 1) % 0x10000
271                    req_adu = messages.TcpAdu(transaction_id=transaction_id,
272                                              device_id=device_id,
273                                              pdu=req_pdu)
274
275                elif self._modbus_type == common.ModbusType.RTU:
276                    req_adu = messages.RtuAdu(device_id=device_id,
277                                              pdu=req_pdu)
278
279                elif self._modbus_type == common.ModbusType.ASCII:
280                    req_adu = messages.AsciiAdu(device_id=device_id,
281                                                pdu=req_pdu)
282
283                else:
284                    raise Exception("invalid modbus type")
285
286                try:
287                    req_adu_bytes = encoder.encode_adu(req_adu)
288                except Exception as e:
289                    future.set_exception(e)
290                    continue
291
292                res_adu_queue = aio.Queue()
293                with self._receive_cbs.register(res_adu_queue.put_nowait):
294
295                    await self._writer.write(req_adu_bytes)
296
297                    if device_id == 0:
298                        res_pdu = messages.Pdu(req_pdu.fc, None)
299                    else:
300                        res_pdu = None
301
302                    while not res_pdu:
303                        async with self.async_group.create_subgroup() as subgroup: # NOQA
304                            res_adu_future = subgroup.spawn(res_adu_queue.get)
305                            await asyncio.wait(
306                                [res_adu_future, future],
307                                return_when=asyncio.FIRST_COMPLETED)
308                            if future.done():
309                                break
310                            res_adu = res_adu_future.result()
311
312                        if self._modbus_type == common.ModbusType.TCP:
313                            if req_adu.transaction_id != res_adu.transaction_id:  # NOQA
314                                mlog.warning("received invalid response "
315                                             "transaction id")
316                                continue
317
318                        if req_adu.device_id != res_adu.device_id:
319                            mlog.warning("received invalid response "
320                                         "device id")
321                            continue
322
323                        res_pdu = res_adu.pdu
324
325                if not future.done():
326                    future.set_result(res_pdu)
327
328        except Exception as e:
329            mlog.error("error in send loop: %s", e, exc_info=e)
330
331        finally:
332            self.close()
333            self._send_queue.close()
334            if future and not future.done():
335                future.set_exception(ConnectionError())
336            while not self._send_queue.empty():
337                _, __, future = self._send_queue.get_nowait()
338                if not future.done():
339                    future.set_exception(ConnectionError())
340
341    async def _receive_loop(self):
342        try:
343            while True:
344                adu = await encoder.read_adu(self._modbus_type,
345                                             encoder.Direction.RESPONSE,
346                                             self._reader)
347                self._receive_cbs.notify(adu)
348
349        except Exception as e:
350            mlog.error("error in read loop: %s", e, exc_info=e)
351
352        finally:
353            self.close()

215    async def _receive_loop(self):
216        try:
217            while True:
218                try:
219                    req_adu = await encoder.read_adu(self._modbus_type,
220                                                     encoder.Direction.REQUEST,
221                                                     self._reader)
222                except (asyncio.IncompleteReadError,
223                        EOFError,
224                        ConnectionError):
225                    break
226
227                device_id = req_adu.device_id
228                fc = req_adu.pdu.fc
229                req = req_adu.pdu.data
230
231                res = await self._process_request(device_id, fc, req)
232                res_pdu = messages.Pdu(fc=fc, data=res)
233
234                if device_id == 0:
235                    continue
236
237                if self._modbus_type == common.ModbusType.TCP:
238                    res_adu = messages.TcpAdu(
239                        transaction_id=req_adu.transaction_id,
240                        device_id=req_adu.device_id,
241                        pdu=res_pdu)
242
243                elif self._modbus_type == common.ModbusType.RTU:
244                    res_adu = messages.RtuAdu(device_id=req_adu.device_id,
245                                              pdu=res_pdu)
246
247                elif self._modbus_type == common.ModbusType.ASCII:
248                    res_adu = messages.AsciiAdu(device_id=req_adu.device_id,
249                                                pdu=res_pdu)
250
251                else:
252                    raise ValueError("invalid modbus type")
253
254                res_adu_bytes = encoder.encode_adu(res_adu)
255                await self._writer.write(res_adu_bytes)
256
257        finally:
258            self.close()
